PART FIRST - CUSTOM ELEMENTS

<element name="x-foo" constructor="XFoo">
  <section>I'm an x-foo!</section>
  <script>
    var section = this.querySelector('section');
    this.register({ prototype: { foo: function() { alert('foo() called'); } } });
  </script>
</element>
      
<x-foo></x-foo>
      
var elem = document.createElement('x-foo');
elem.foo(); // alert 'foo() called'.
var elem = new XFoo();
document.body.appendChild(elem);

PART ORIGINAL - MODEL DRIVEN VIEWS

"Seven-six-two millimeter. Full metal jacket." - Private Gomer Pyle

<element name="x-foo-shadow" constructor="XFooShadow">
  <template>
    <style> @host { * { display: block; background: #ffcc00; ... } } </style>
    <section>I'm an x-foo-shadow. Gots me some Shadow DOM!</section>
  </template>
  <script>
    var template = this.querySelector('template');
    this.register({ // this == <element>
      prototype: { readyCallback: function() {
         this.createShadowRoot().appendChild(template.content.cloneNode(true));
      }}
    }});
  </script>
</element>
